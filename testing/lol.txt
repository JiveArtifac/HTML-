<script>
function toggleTheme() {
    document.body.classList.toggle("dark");
}

/* Calculator */
function calc(op) {
    let a = +document.getElementById("a").value;
    let b = +document.getElementById("b").value;

    let r = eval(a + op + b);
    document.getElementById("calcResult").innerText = r;
}

/* Tax */
function taxCalc() {
    let i = +document.getElementById("income").value;
    let t = i <= 5000 ? i*0.05 : i <= 20000 ? i*0.1 : i*0.15;
    document.getElementById("taxResult").innerText = t;
}

/* BMI */
function bmiCalc() {
    let w = +document.getElementById("weight").value;
    let h = +document.getElementById("height").value / 100;
    document.getElementById("bmiResult").innerText = (w/(h*h)).toFixed(2);
}

/* Convert */
function convert() {
    let v = +document.getElementById("value").value;
    let t = document.getElementById("type").value;

    let r =
        t=="cm-m"?v/100:
        t=="m-cm"?v*100:
        t=="m-km"?v/1000:
        v*1000;

    document.getElementById("convertResult").innerText = r;
}
</script>
